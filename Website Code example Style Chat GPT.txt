  <script>
    // small interactive helpers
    function handleSubmit(e){
      e.preventDefault();
      const name = document.getElementById('name').value || 'friend';
      const note = document.getElementById('note');
      note.textContent = `Thanks, ${name}! This demo does not submit data — it's local only.`;
      // Add a tiny animation to the button to show success
      const btn = document.querySelector('button.cta');
      btn.classList.add('pulse');
      setTimeout(()=>btn.classList.remove('pulse'), 1000);
      return false;
    }

    function fillExample(){
      document.getElementById('name').value = 'Alex Example';
      document.getElementById('email').value = 'alex@example.com';
      document.getElementById('message').value = 'Loving the single-file template — thanks!';
      document.getElementById('note').textContent = 'Example content filled.';
    }
